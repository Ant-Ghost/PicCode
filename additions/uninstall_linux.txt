#!/bin/bash

echo -e "\n\n***WARNING***\n"
echo -e "Do you want to remove Pillow?(y/n)"

read -n 1
if [[ "${REPLY}" == "Y" ]] || [[ "${REPLY}" == "y" ]]
then
    echo -e "\n\nRunning command: python3 -m pip uninstall Pillow\n"
    echo y | python3 -m pip uninstall Pillow
fi

echo -e "\n\n***WARNING***\n"
echo -e "Do you want to remove python3-pip?(y/n)"

read -n 1
if [[ "${REPLY}" == "Y" ]] || [[ "${REPLY}" == "y" ]]
then
    echo -e "\n\nRunning command: sudo apt-get uninstall -y python3-pip\n"
    sudo apt-get uninstall -y python3-pip
fi

echo -e "\n\n***WARNING***\n"
echo -e "Do you want to remove python3?(y/n)"

read -n 1
if [[ "${REPLY}" == "Y" ]] || [[ "${REPLY}" == "y" ]]
then
    echo -e "\n\nRunning command: sudo apt-get uninstall -y python3\n"
    sudo apt-get uninstall -y python3
fi

echo -e "\n\nUninstallation done...\n\n"

rm run.sh uninstall.sh

exit 0
